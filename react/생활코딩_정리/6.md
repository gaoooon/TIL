# 생활코딩

## event

### 이벤트는 어떤사건을 발생시키는 것을 말한다

### 이벤트 사용 예

```jsx
import "./App.css";

function Header(props) {
  return (
    <header>
      <h1>
        <a
          href=""
          // a태그가 클릭을 당했을때 실행된다
          onClick={(event) => {
            event.preventDefault(); // a태그가 클릭 됬을때 실행되는 기본동작을 막아주는 함수
            props.onChangeMode(); // props 안에 있는 onChangeMode() 함수를 실행시킨다
          }}
        >
          {props.title}
        </a>
      </h1>
    </header>
  );
}

function Nav(props) {
  const list = [];
  for (let i = 0; i < props.topics.length; i++) {
    let t = props.topics[i];
    list.push(
      // 자동으로 생성된 태그의 경우에는 태그에 key값을 주어서 react가 태그를 추적할때 도움을준다
      <li key={t.id}>
        <a
          id={t.id}
          href={"/read/" + t.id}
          onClick={(event) => {
            event.preventDefault();
            props.onChangeMode(event.target.id);
          }}
        >
          {t.title}
        </a>
      </li>
    );
  }
  return (
    <nav>
      <ol>{list}</ol>
    </nav>
  );
}

function Article() {
  return (
    <article>
      <h2>Welcome</h2>
      Hello, WEB
    </article>
  );
}

function App() {
  const topics = [
    { id: 1, title: "html", body: "html is ..." },
    { id: 2, title: "css", body: "css is ..." },
    { id: 3, title: "js", body: "js is ..." },
  ];

  return (
    <div>
      <Header
        title="hello, react"
        onChangeMode={() => alert("header")}
      ></Header>
      <Nav
        topics={topics}
        onChangeMode={(id) => {
          alert(id);
        }}
      ></Nav>
      <Article></Article>
    </div>
  );
}
export default App;
```
